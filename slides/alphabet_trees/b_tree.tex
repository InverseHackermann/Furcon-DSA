\frame{
    \frametitle{B Tree\footnote{Bayer, R.; McCreight, E. (July 1970).}}
  \begin{itemize}
    \item Generalising binary search trees to $>2$ children
    
    \medskip
    
    \item Internal nodes hold both data and pointers to children nodes in sorted order
    
    \medskip
    
    \item Forms the basis of many filesystem structures 
    
    \begin{itemize}
        \item Apple HFS+, Microsoft NTFS, Linux ext4
    \end{itemize}
  \end{itemize}
}
