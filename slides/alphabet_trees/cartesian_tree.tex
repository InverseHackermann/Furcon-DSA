  \frametitle{Cartesian Tree\footnote{Vuillemin, Jean (1980)}}
  \begin{columns}
    \begin{column}{0.5\textwidth}
    \begin{itemize}

    \item Binary tree derived from a sequence of distinct numbers
    
    \medskip
    
    \item Take the smallest number as root, recursively build on left and right

    \medskip
    
        \item Turns range minimum queries into lowest common ancestor queries, allowing for constant time $O(1)$ range minimum queries
    \end{itemize}
