  \frametitle{Gomory-Hu Tree\footnote{Gomory, R. E.; Hu, T. C. (1961).}}
  \begin{itemize}
    \item Represents the min-cut capacities of all pairs of vertices in a flow network

    \medskip
    
    \item Min-weight along a path in the tree corresponds to the min-cut capacity between the endpoints of the path.
  \end{itemize}
