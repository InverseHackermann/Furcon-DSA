\frame{
  \frametitle{Order Statistics Tree}
  \begin{itemize}
    \item A tree that can efficiently find the following
    \begin{itemize}
        \item given index $i$, $i$-th smallest element in the set
        \item given element $x$, number of elements smaller than $x$ in the set
    \end{itemize}
  \end{itemize}
