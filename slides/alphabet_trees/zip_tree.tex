    \begin{column}{0.4\textwidth}
    \begin{itemize}
    \item Randomized BST with expected $O(\log n)$ depth! 

    \medskip

    \item Insert takes a section of the insertion path and unzips it into two children

    \medskip

    \item Delete takes the inner paths of the children and zips them together
    \end{itemize}
