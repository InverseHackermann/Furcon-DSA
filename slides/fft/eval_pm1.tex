f(x)&=a_0+a_1x+a_2x^2+a_3x^3+\dots+a_{n-2}x^{n-2}+a_{n-1}x^{n-1}\\
\WhenNotMeasuring{\pause}\\
f(1)&=a_0+a_1+a_2+a_3+\dots+a_{n-2}+a_{n-1}\\
f(-1)&=a_0-a_1+a_2-a_3+\dots+a_{n-2}-a_{n-1}\\
\WhenNotMeasuring{\pause}\\
f(1)&=(a_0+a_2+\dots+a_{n-2})+(a_1+a_3+\dots+a_{n-1})\\
f(-1)&=(a_0+a_2+\dots+a_{n-2})-(a_1+a_3+\dots+a_{n-1})\\
\end{align*}
